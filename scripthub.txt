-- Notificación
local function showNotification(message, isError)
    local n = Instance.new("TextLabel", main)
    n.Text = message
    n.Size = UDim2.new(1, 0, 0, 30)
    n.Position = UDim2.new(0, 0, 1, 0)
    n.Font = Enum.Font.SourceSansBold
    n.TextSize = 16
    n.BackgroundColor3 = isError and Color3.fromRGB(180, 40, 40) or green
    n.TextColor3 = Color3.new(1, 1, 1)
    n.BorderSizePixel = 0

    game:GetService("TweenService"):Create(n, TweenInfo.new(0.5), {Position = UDim2.new(0, 0, 1, -30)}):Play()
    game:GetService("Debris"):AddItem(n, 3)
    wait(2)
    game:GetService("TweenService"):Create(n, TweenInfo.new(0.5), {Position = UDim2.new(0, 0, 1, 0)}):Play()
end

-- Crear botones
local count = 0
for name, func in pairs(scripts) do
    local btn = Instance.new("TextButton", scroll)
    btn.Text = name
    btn.Size = UDim2.new(1, -10, 0, 40)
    btn.BackgroundColor3 = Color3.fromRGB(30, 90, 30)
    btn.BorderColor3 = green
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = Enum.Font.SourceSans
    btn.TextSize = 16

    btn.MouseEnter:Connect(function()
        btn.BackgroundColor3 = green
    end)
    btn.MouseLeave:Connect(function()
        btn.BackgroundColor3 = Color3.fromRGB(30, 90, 30)
    end)

    btn.MouseButton1Click:Connect(function()
        local success, err = pcall(func)
        if success then
            showNotification("¡" .. name .. " ejecutado!")
        else
            showNotification("Error ejecutando " .. name, true)
            warn(err)
        end
    end)
    
    count += 1
end

scroll.CanvasSize = UDim2.new(0, 0, 0, count * 45)
